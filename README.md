ARRAY METHOD 
// PUSH//
let num = [11,12,14,15,16]
num.push(19)
console.log(num)
[11,13,14,15,16,19]

//UNSHIFT//
num.unshift(10)
console.log(num)
[10,11,13,14,15,16,19]

//POP//
num.pop()
[10,11,13,14,15,16,]

//SHIFT//
num.shift()
console.log(num)
[11,13,14,15,16,19]

//SPLICE//
num.splice(2)
console.log(num)
[14,15,16,19]

//SLICE//
let num = [11,12,14,15,16]
num.slice(11,15)
console.log(num)
 [12,14,15,]

 //FLITER//
 let num = [11,12,14,15,16]
 num.fliter(n => n%2==1)
 console.log(num)
 [12,14,16]

 //MAP//
 let num = [2,4,7,8]
 num.map(n=> n*2)
 console.log(num)
 [2,8,14,16]

 //INDEX0F//
 let num = [2,3,6,8,9,]
 num.indexof(8)
 console.log(num)
 [3]
